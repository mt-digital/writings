\section{History of PDE}
\begin{frame}{Hebbian rule}

    Suppose \(m\) orthogonal vectors \(\mathbf{a}(1) \ldots \mathbf{a}(m)\)

    Weights matrix is written \footfullcite{Rumelhart1986a}

    \begin{equation}
        w_{ij} = \sum_{p=1}^m \mathbf{a}_i(p) \mathbf{a}_j(p)
    \end{equation}

\end{frame}
